### the function

``` r
sim_regression = function(beta_1, beta_0 = 2,n=30) {
  
  sim_data = tibble(
    x = rnorm(n, mean = 0, sd = 1),
    y = beta_0 + beta_1 * x + rnorm(n, 0, 50)
  )
  
  ls_fit = lm(y ~ x, data = sim_data)
  
  c(
    tidy(ls_fit)[2,2],
    tidy(ls_fit)[2,5]
  )
  
}
```
